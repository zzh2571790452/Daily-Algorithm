二叉搜索树的性质:左子树小于节点值,右子树大于节点值

即需要递归左节点,后节点值,后右节点

又由于需要头节点与尾节点相连,故利用pre与cur两指针
```
class Solution {
public:
    Node* treeToDoublyList(Node* root) {
        if(root == nullptr) return nullptr;
        dfs(root);
        head->left = pre;
        pre->right = head;
        return head;
    }
private:
    Node *pre, *head;
    void dfs(Node* cur) {
        if(cur == nullptr) return;  //越出叶节点时完成递
        dfs(cur->left); //递至左下角节点
        if(pre != nullptr) pre->right = cur;    //将左节点右侧连接右节点
        else head = cur;    //找到头节点
        cur->left = pre;    //将右节点左侧连接左节点
        pre = cur;  //更新pre
        dfs(cur->right);    //递归右子树
    }
};
```
