class Solution {
public:
    bool checkSymmetricTree(TreeNode* root) {
        return root==nullptr||recur(root->left,root->right);    //单独拿出root为空指针时的情况
    }
private:
    bool recur(TreeNode* l,TreeNode* r){
        if(l==nullptr&&r==nullptr) return true; //当两侧均为空时,仍成镜面,且为越出叶节点的标志
        if(l==nullptr||r==nullptr||l->val!=r->val) return false;    //由于经过判断之后,l和r不可能均为空,故值不相等或有一个为空指针时返回false
        return recur(l->left,r->right)&&recur(l->right,r->left);    递归
    }
};
