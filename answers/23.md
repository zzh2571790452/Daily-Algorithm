class Solution {
public:
    vector<int> toggleLightBulbs(vector<int>& bulbs) {
        unordered_map<int,int> map;
        vector<int> res;
        for(int i : bulbs){
            map[i]++;
        }
        for(auto& pair:map){
            if(pair.second%2!=0)    //判断开关情况
                res.push_back(pair.first);
        }
        sort(res.begin(),res.end());    //保证升序排列
        return res;
    }
};
