class Solution {
public:
    int countTarget(vector<int>& scores, int target) {
        return recur(scores,target)-recur(scores,target-1); //天才,用target-1使得可以用封装函数解决,而非求左右边界
    }
private:
    int recur(vector<int>& scores,int target){
        int i = 0;
        int j = scores.size()-1;
        while(i<=j){
            int m = (i+j)/2;
            if(scores[m]<=target) i=m+1;
            else j=m-1;
        }
        return i;
    }
};
