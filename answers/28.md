class Codec {
public:
    void rserialize(TreeNode* root, string& str) {
        if (root == nullptr) {  //标记空指针
            str += "None,";
        } else {    //存储指针值
            str += to_string(root->val) + ",";
            rserialize(root->left, str);
            rserialize(root->right, str);
        }
    }

    string serialize(TreeNode* root) {
        string ret;
        rserialize(root, ret);
        return ret;
    }

    TreeNode* rdeserialize(list<string>& dataArray) {
        if (dataArray.front() == "None") {  //放置空指针
            dataArray.erase(dataArray.begin());
            return nullptr;
        }

        TreeNode* root = new TreeNode(stoi(dataArray.front())); //利用stoi将字符串转换为数字
        dataArray.erase(dataArray.begin());
        root->left = rdeserialize(dataArray);   //左右节点建立联系
        root->right = rdeserialize(dataArray);
        return root;
    }

    TreeNode* deserialize(string data) {
        list<string> dataArray;
        string str;
        for (auto& ch : data) {
            if (ch == ',') {    //当为','结算转化
                dataArray.push_back(str);
                str.clear();
            } else {    //不为','时,存储字符
                str.push_back(ch);
            }
        }
        if (!str.empty()) { //将字符串存入list
            dataArray.push_back(str);
            str.clear();
        }
        return rdeserialize(dataArray);
    }
};
