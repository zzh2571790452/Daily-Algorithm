>模拟栈

由选定最左边的一组相同相邻数据合并,联想到**栈**

由于需要比较最后元素一个元素和倒数第二个元素,故需利用数组模拟栈

```
class Solution {
public:
    vector<long long> mergeAdjacent(vector<int>& nums) {
        vector<long long> st;
        for (int x : nums) {
            st.push_back(x);
            while (st.size() > 1 && st.back() == st[st.size() - 2]) {   //当倒数1 2元素相同时合并
                st.pop_back();
                st.back() *= 2;
            }
        }
        return st;
    }
};
```
