class CQueue {
public:
    stack<int> A,B;
    CQueue() {}
    
    void appendTail(int value) {
        A.push(value);
    }
    
    int deleteHead() {
        if(!B.empty()){ //已将A中元素填入B时,进行这个语句
            int tmp = B.top();
            B.pop();
            return tmp;
        }
        if(A.empty()) return -1;    //未填入时A即为空或者填入后B耗尽元素执行这一语句
        while(!A.empty()){  //填入A中元素
            int tmp = A.top();
            A.pop();
            B.push(tmp);
        }
        int tmp = B.top();  //首次时会进行这一语句
        B.pop();
        return tmp;
    }
};

/**
 * Your CQueue object will be instantiated and called as such:
 * CQueue* obj = new CQueue();
 * obj->appendTail(value);
 * int param_2 = obj->deleteHead();
 */
