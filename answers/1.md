class Solution {
public:
    vector<int> statisticalResult(vector<int>& arrayA) {
        int len = arrayA.size();
        if(len == 0) return {};     //数组长度为0时直接返回
        vector<int> arrayB(len, 1);     
        arrayB[0] = 1;
        int tmp = 1;
        for(int i = 1; i < len; i++) {
            arrayB[i] = arrayB[i - 1] * arrayA[i - 1];      //计算下三角
        }
        for(int i = len - 2; i >= 0; i--) {
            tmp *= arrayA[i + 1];               
            arrayB[i] *= tmp;       //计算上三角
        }
        return arrayB;
    }
};

        vector<int> arrayB;

        int all = 1;
        int hasZero = 0;
        for(int i = 0;i<n;++i){
            if(hasZero==2) break;   //当有两个0时,arrayB各项均为0
            if(arrayA[i]==0){
                hasZero++;
                continue;
            }
            all*=arrayA[i];
        }
        for(int i = 0;i<n;++i){
            if(hasZero==1){                 //只有一个0的情况
                if(arrayA[i]==0) 
                    arrayB.push_back(all);
                else
                    arrayB.push_back(0);
            }
            else if(hasZero==2){        //有两个0时
                arrayB.push_back(0);
            }
            else    //不含0时               
                arrayB.push_back(all/arrayA[i]);
        }
        return arrayB;
