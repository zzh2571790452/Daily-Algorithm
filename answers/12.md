确定一个非严格递减的队列和滑动窗口

```
class Solution {
public:
    vector<int> maxAltitude(vector<int>& heights, int limit) {
        deque<int> d;
        vector<int> res;
        int n = heights.size();
        if(n==0||limit==0) return {};
        for(int i = 1-limit,j = 0;j<n;++i,++j){
            if(i>0&&heights[i-1]==d.front()) d.pop_front(); //当移出滑动窗口的元素为队列首元素时,删除该元素
            while(!d.empty()&&d.back()<heights[j]) d.pop_back();    //每次循环中更新非严格递减序列
            d.push_back(heights[j]);
            if(i>=0) res.push_back(d.front());  //填入元素
        }
        return res;
    }
};
```
