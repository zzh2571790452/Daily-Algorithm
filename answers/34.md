扆梑唅蛌萸撈褫
```
class Solution {
public:
    int inventoryManagement(vector<int>& stock) {
        int i = 0,j = stock.size()-1;
        while(i<j){
            int m = (i+j)/2;
            if(stock[m]>stock[j]) i=m+1;
            else if(stock[m]<stock[i]) j=m;
            else j--;
        }
        return stock[i];
    }
};
```
