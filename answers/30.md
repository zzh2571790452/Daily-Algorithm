class Solution {
    static constexpr int fac[] = {1, 1, 2, 6, 24, 120, 720, 5040, 40320, 362880};   //0~9的阶乘

public:
    bool isDigitorialPermutation(int n) {
        int sum_fac = 0;
        array<int, 10> cnt{};   //最多存储10个int类型元素的容器
        for (; n > 0; n /= 10) {    //记录各个元素数量
            int d = n % 10;
            sum_fac += fac[d];
            cnt[d]++;
        }

        for (; sum_fac > 0; sum_fac /= 10) {    //判断与原数各个元素数量是否相等
            cnt[sum_fac % 10]--;
        }

        // cnt[i] == 0
        return cnt == array<int, 10>();
    }
};
