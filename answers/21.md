class Solution {
public:
    int findTargetNode(TreeNode* root, int cnt) {
        this->cnt = cnt;    //传参至private
        dfs(root);
        return res;
    }
private:
    int res, cnt;
    void dfs(TreeNode* root) {
        if(root == nullptr) return;
        dfs(root->right);
        if(cnt == 0) return;    //说明输入cnt不合法,越界
        if(--cnt == 0) res = root->val; //返回第n大成员
        dfs(root->left);
    }
};
