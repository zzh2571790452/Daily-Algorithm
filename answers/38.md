class Solution {
public:
    vector<vector<int>> pathSum(TreeNode* root, int sum) {
        recur(root, sum);
        return res;
    }
private:
    vector<vector<int>> res;
    vector<int> path;
    void recur(TreeNode* root, int tar) {
        if(root == nullptr) return; //老岺係숭
        path.push_back(root->val);  //닸흙쨌쓺
        tar -= root->val;
        if(tar == 0 && root->left == nullptr && root->right == nullptr) //털뙤찮璃狼헹
            res.push_back(path);
        recur(root->left, tar); //깁저
        recur(root->right, tar);
        path.pop_back();    //뫘劤쨌쓺
    }
};
