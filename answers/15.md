题目要求按层打印二叉树，即二叉树的 广度优先遍历 ，其通常借助 队列 的先入先出特性来实现。
```
class Solution {
public:
    vector<int> decorateRecord(TreeNode* root) {
        vector<int> res;
        if(!root) return res;
        queue<TreeNode *> que;  //存储同一层级的树的节点
        que.push(root); 
        while(!que.empty()){
            TreeNode* node = que.front(); //存储当前节点指针
            que.pop();  //将已存储的元素移除队列
            res.push_back(node->val);   //将值填入res
            if(node->left) que.push(node->left);
            if(node->right) que.push(node->right);
        }
        return res;
    }
};
```
